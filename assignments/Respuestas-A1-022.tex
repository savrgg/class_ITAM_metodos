$$\sigma^2_{\bar{y}} =  \frac{\sigma^2}{n}= 1/9 $$

\text{Tenemos que:}

%$$ P(\abs{\bar{y} (-) \mu} \leq 0.3) = $$

$$ P(-0.3 \leq (bar{y}-\mu) \leq 0.3)

$$

%$$ P(\abs{\bar{y}-\mu} \leq 0.3)  = $$

$$P(-0.9 \leq Z \leq 0.9)  = 1- 2P(Z \leq 0.9) = 1- 2(0.1841) = 0.6318
$$
